```shell
echo $0  # the current script file name
echo $n  # the script n-th input param
echo $#  # the num of the script input params
echo $@  # all arguments passed to function or script
echo $*  # all arguments passed to function or script
echo $$  # the process id of the current shell
echo $!  # the last job placed in background
echo $?  # the exit status of the last command executed
```
